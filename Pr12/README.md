## Практическая работа 12. Применение параллельных вычислений в .NET
## Пример кода
using System;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        Console.WriteLine("Вывод чисел от 1 до 20 с использованием Parallel.For:");

        Parallel.For(1, 21, i =>
        {
            Console.WriteLine($"Число {i}, поток {Task.CurrentId}");
        });

        Console.WriteLine("Завершение работы программы.");
    }
}


## Пояснения:

Parallel.For(1, 21, i => { ... }) — выполняет итерации цикла параллельно.

Task.CurrentId показывает идентификатор потока, выполняющего текущую итерацию.

В отличие от обычного for, порядок вывода чисел может быть неупорядоченным, так как итерации выполняются параллельно.
