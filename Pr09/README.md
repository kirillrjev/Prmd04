## Практическая работа 9. Создание и управление потоками в .NET
## Пример кода
using System;
using System.Threading;

class Program
{
    static void Main()
    {
        // Создаём поток для вывода чисел от 1 до 20
        Thread numberThread = new Thread(() =>
        {
            for (int i = 1; i <= 20; i++)
            {
                Console.WriteLine(i);
                Thread.Sleep(100); // небольшая пауза для наглядности
            }
        });

        // Запуск потока
        numberThread.Start();

        // Ожидание завершения потока
        numberThread.Join();

        Console.WriteLine("Поток завершён.");
    }
}


## Пояснения:

Thread t = new Thread(() => { ... }); — создаёт новый поток с лямбда-выражением.

numberThread.Start() — запускает выполнение потока.

numberThread.Join() — основной поток ждёт завершения нового потока, чтобы продолжить работу.

Thread.Sleep(100) — имитация небольшого времени работы потока, делает вывод более наглядным.
